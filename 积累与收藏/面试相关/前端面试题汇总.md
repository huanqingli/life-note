    _name: 'Bob',
    getName: () => this._name

    this.age = age
    this.growOld = function() {
        this.age++;
    }
}

setTimeout(person.growOld,1000);

setTimeout(function() { console.log(person.age); },2000);

- b-7: 函数式综合: 实现 add(2)(3)(4) //输出9
```js
function add(x) {
    var sum = x;
    var tmp = function (y) {
        sum = sum + y;
        return tmp;
    };
    tmp.toString = function () {
        return sum;
    };
    return tmp;
}
console.log(add(1)(2)(3));  //6
console.log(add(1)(2)(3)(4));   //10
```
